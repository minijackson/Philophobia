001package gameplay;
002
003import main.Philophobia;
004import debug.CliOptions;
005import window.Window;
006import world.World;
007import gameplay.ai.AI;
008
009/**
010 * Class handling the game play of this game
011 * <p>
012 * It handle the creation of worlds, the movements
013 * of the main character and all the other characters,
014 * (including the constraints), the interactions between
015 * the main character and the other characters or the
016 * scenery
017 * <p>
018 * The GamePlay class also controls the user interface mode
019 * (loading screen, win, fail, etcâ€¦)
020 * <p>
021 * This class is the equivalent of the controller in the
022 * MVC paradigm.
023 * @see world.World
024 */
025public class GamePlay {
026
027        /**
028         * World currently being played
029         */
030        protected World currentWorld;
031
032        /**
033         * Name of the world currently being played
034         * <p>
035         * It is used to detect in which world the
036         * player is and where are the other worlds
037         */
038        protected String currentWorldName;
039
040        /**
041         * The game window
042         */
043        protected Window window;
044
045        /**
046         * Malicious artificial intelligence
047         * driving the player mad
048         */
049        protected AI artificialIntelligence;
050
051        /**
052         * Constructor of the GamePlay class
053         * @param cliOptions Opions for the game given via the command line
054         */
055        public GamePlay(CliOptions cliOptions) {
056                Philophobia.getVerbose().information("Creating GamePlay class", "gameplay/Gameplay.java", "GamePlay.GamePlay()");
057
058                // The default world is the Temperate Broadleaf world
059                currentWorldName = "Temperate Broadleaf";
060                currentWorld = new World("temperatebroadleaf");
061
062                window = new Window();
063
064                try {
065                        // Sleep during the length of the animation
066                        Thread.sleep(2000);
067                } catch(InterruptedException e) {
068                        Philophobia.getVerbose().warning("Sleep interrupted: " + e.getMessage(), "gameplay/GamePlay.java", "GamePlay.GamePlay(CliOptions)");
069                }
070
071                artificialIntelligence = new AI(currentWorld);
072
073                window.getUserInterface().setDisplayedWorld(currentWorld);
074                window.setGameState();
075        }
076
077};
