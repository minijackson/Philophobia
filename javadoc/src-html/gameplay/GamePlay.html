001package gameplay;
002
003import main.Philophobia;
004import debug.CliOptions;
005import window.Window;
006import world.World;
007
008/**
009 * Class handling the game play of this game
010 * <p>
011 * It handle the creation of worlds, the movements
012 * of the main character and all the other characters,
013 * (including the constraints), the interactions between
014 * the main character and the other characters or the
015 * scenery
016 * <p>
017 * The GamePlay class also controls the user interface mode
018 * (loading screen, win, fail, etcâ€¦)
019 * @see world.World
020 */
021public class GamePlay {
022
023        /**
024         * World currently being played
025         */
026        protected World currentWorld;
027
028        /**
029         * Name of the world currently being played
030         * <p>
031         * It is used to detect in which world the
032         * player is and where are the other worlds
033         */
034        protected String currentWorldName;
035
036        /**
037         * The game window
038         */
039        protected Window window;
040
041        /*
042         * Malicious artificial intelligence
043         * driving the player mad
044         */
045//      protected AI artificialIntelligence;
046
047        /**
048         * Constructor of the GamePlay class
049         * @param cliOptions Opions for the game given via the command line
050         */
051        public GamePlay(CliOptions cliOptions) {
052                Philophobia.getVerbose().information("Creating GamePlay class", "gameplay/Gameplay.java", "GamePlay.GamePlay()");
053
054                // The default world is the Temperate Broadleaf world
055                currentWorldName = "Temperate Broadleaf";
056                currentWorld = new World(Philophobia.getWorldFilePrefix() + "temperatebroadleaf.world");
057
058                window = new Window();
059
060                try {
061                        // Sleep during the length of the animation
062                        Thread.sleep(2000);
063                } catch(InterruptedException e) {
064                        Philophobia.getVerbose().warning("Sleep interrupted: " + e.getMessage(), "gameplay/GamePlay.java", "GamePlay.GamePlay(CliOptions)");
065                }
066
067                window.getUserInterface().setDisplayedWorld(currentWorld);
068                window.setGameState();
069        }
070
071};
