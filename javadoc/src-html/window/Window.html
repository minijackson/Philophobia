001package  window;
002
003import main.Philophobia;
004import window.UI;
005
006import javax.swing.JFrame;
007import java.awt.Toolkit;
008
009public class Window extends JFrame {
010
011        /**
012         * User interface which handle the graphics inside the window
013         * @see window.UI
014         */
015        private UI userInterface;
016
017        /**
018         * Constructor of the Window class
019         */
020        public Window() {
021                Philophobia.getVerbose().calls("Creating Window class", "org/window/Window.java", "Window.Window()");
022
023                this.setTitle("Philophobia");
024                this.setSize(((int) Toolkit.getDefaultToolkit().getScreenSize().getWidth()), ((int) Toolkit.getDefaultToolkit().getScreenSize().getHeight()));
025                this.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
026                
027                // Fullscreen mode
028                this.setExtendedState(this.getExtendedState() | JFrame.MAXIMIZED_BOTH);
029                this.setUndecorated(true);
030
031                userInterface = new UI(this.getHeight(), this.getWidth());
032                this.setContentPane(userInterface);
033
034                this.setVisible(true);
035        }
036
037};
