001package  window;
002
003import main.Philophobia;
004import window.ui.UI;
005import window.topbar.TopBar;
006
007import javax.swing.JFrame;
008import javax.swing.JPanel;
009import java.awt.Toolkit;
010import java.awt.BorderLayout;
011
012public class Window extends JFrame {
013
014        /**
015         * User interface which handle the game graphics inside the window
016         * @see window.ui.UI
017         */
018        private UI userInterface;
019
020        private TopBar topbar;
021
022        /**
023         * Constructor of the Window class
024         */
025        public Window() {
026                Philophobia.getVerbose().calls("Creating Window class", "window/Window.java", "Window.Window()");
027
028                this.setTitle("Philophobia");
029                this.setSize(((int) Toolkit.getDefaultToolkit().getScreenSize().getWidth()), ((int) Toolkit.getDefaultToolkit().getScreenSize().getHeight()));
030                this.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
031                
032                // Fullscreen mode
033                this.setExtendedState(this.getExtendedState() | JFrame.MAXIMIZED_BOTH);
034                this.setUndecorated(true);
035
036                this.setLayout(new BorderLayout());
037        
038                userInterface = new UI(this.getHeight(), this.getWidth());
039                this.getContentPane().add(userInterface, BorderLayout.CENTER);
040
041                TopBar topbar = new TopBar();
042                this.getContentPane().add(topbar, BorderLayout.PAGE_START);
043
044/*
045                this.setContentPane(userInterface);
046*/
047                this.pack();
048                this.setVisible(true);
049
050        }
051
052};
