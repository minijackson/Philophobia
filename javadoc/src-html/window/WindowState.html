001package  window;
002
003import main.Philophobia;
004import window.ui.UI;
005import window.topbar.TopBar;
006
007import javax.swing.JFrame;
008import javax.swing.JPanel;
009import java.awt.Toolkit;
010import java.awt.BorderLayout;
011
012public class Window extends JFrame {
013
014        /**
015         * Window's current state
016         * @see window.WindowState
017         */
018        protected WindowState windowState;
019
020        /**
021         * Window top bar
022         */
023        protected TopBar topbar;
024
025        /**
026         * User interface which handle the game graphics inside the window
027         */
028        protected UI userInterface;
029
030        /**
031         * Constructor of the Window class
032         */
033        public Window() {
034                Philophobia.getVerbose().calls("Creating Window class", "window/Window.java", "Window.Window()");
035
036                this.setTitle("Philophobia");
037                this.setSize(((int) Toolkit.getDefaultToolkit().getScreenSize().getWidth()), ((int) Toolkit.getDefaultToolkit().getScreenSize().getHeight()));
038                this.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
039                
040                // Fullscreen mode
041                this.setExtendedState(this.getExtendedState() | JFrame.MAXIMIZED_BOTH);
042                this.setUndecorated(true);
043
044                this.setLayout(new BorderLayout());
045
046                windowState = WindowState.FirstLoading;
047
048                userInterface = new UI(this.getHeight(), this.getWidth());
049//              this.getContentPane().add(userInterface, BorderLayout.CENTER);
050
051                topbar = new TopBar();
052                this.getContentPane().add(topbar, BorderLayout.PAGE_START);
053                this.pack();
054                this.setVisible(true);
055
056        }
057
058        public void setLoadingState() {
059                Philophobia.getVerbose().information("Setting window's Loading mode", "window/Window.java", "Window.setLoadingState()");
060                topbar.hidePauseButton();
061                windowState = WindowState.LoadingScreen;
062        }
063
064        public void setGameState() {
065                Philophobia.getVerbose().information("Setting window's Game mode", "window/Window.java", "Window.setGameState()");
066                topbar.showPauseButton();
067                this.getContentPane().add(userInterface, BorderLayout.CENTER);
068                windowState = WindowState.Game;
069        }
070
071        public void setFailState() {
072                Philophobia.getVerbose().information("Setting window's Fail mode", "window/Window.java", "Window.setFailState()");
073                topbar.hidePauseButton();
074                windowState = WindowState.Fail;
075        }
076
077        public void setWinState() {
078                Philophobia.getVerbose().information("Setting window's Win mode", "window/Window.java", "Window.setWindState()");
079                topbar.hidePauseButton();
080                windowState = WindowState.Win;
081        }
082
083        public UI getUserInterface() {
084                return userInterface;
085        }
086
087};
088
089/**
090 * Enumeration of the possible states of the displayed graphics
091 * <p>
092 * The state can be the "in game" state, a loading screen,
093 * the "win" or "fail" screen
094 * @see Window
095 * @see #FirstLoading
096 * @see #Game
097 * @see #LoadingScreen
098 * @see #Win
099 * @see #Fail
100 */
101enum WindowState {
102
103        /**
104         * The FirstLoading state correspond to
105         * the state where the window has just
106         * been called and a special animation
107         * is displayed
108         */
109        FirstLoading,
110
111        /**
112         * The Game state correspond to the state
113         * in which the player is playing the game
114         */
115        Game,
116
117        /**
118         * The LoadingScreen state correspond to
119         * the state where a loading animation
120         * is displayed
121         */
122        LoadingScreen,
123
124        /**
125         * The Win state correspond to the state
126         * where the player has won the game and
127         * a congratulation screen is displayed
128         */
129        Win,
130
131        /**
132         * The Fail state correspond to the state
133         * where the player has lost the game
134         * and a condolence screen is displayed
135         */
136        Fail
137
138};
