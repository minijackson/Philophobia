001package window;
002
003import main.Philophobia;
004import world.World;
005
006import java.awt.Graphics;
007import javax.swing.JPanel;
008
009/**
010 * Class used to handle the window's user interface
011 * <p>
012 * The UI is composed most of the time of two parts :
013 * the top bar and the gameplay part (which basically
014 * contain the world)
015 * <p>
016 * The UI can be also in "Loading mode" which consist
017 * of a loading animation screen
018 */
019public class UI extends JPanel {
020
021        /**
022         * World displayed inside the window
023         */
024        protected World displayedWorld;
025
026        /**
027         * Window height
028         */
029        protected int windowHeight;
030
031        /**
032         * Window width
033         */
034        protected int windowWidth;
035
036        public UI(int windowHeight, int windowWidth) {
037                // We call the parent's constructor
038                super();
039
040                Philophobia.getVerbose().calls("Creating UI class", "org/window/UI.java", "UI.UI(int, int)");
041
042                this.windowHeight = windowHeight;
043                this.windowWidth = windowWidth;
044
045        }
046
047        /**
048         * Function called when the program ask to paint the graphics
049         */
050        public void paintComponent(Graphics g) {
051                Philophobia.getVerbose().calls("painting components", "org/window/UI.java", "UI.paintComponent(Graphics)");
052                
053                g.drawRect(0,0,10, windowWidth);
054
055                displayedWorld = new World("test", g, windowHeight - 10, windowWidth);
056        }
057
058};
