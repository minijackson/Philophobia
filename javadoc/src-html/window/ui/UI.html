001package window.ui;
002
003import main.Philophobia;
004import world.World;
005import world.scenery.Scenery;
006import world.scenery.Tree;
007
008import java.awt.Graphics;
009import javax.swing.JPanel;
010
011/**
012 * Class used to handle the window's user interface
013 * 
014 * This class can be in different states
015 */
016public class UI extends JPanel {
017
018        /**
019         * World displayed inside the window
020         */
021        protected World displayedWorld;
022
023        /**
024         * Window height
025         */
026        protected int windowHeight;
027
028        /**
029         * Window width
030         */
031        protected int windowWidth;
032
033        /**
034         * UI class constructor
035         * @param windowHeight Height of the window
036         * @param windowWidth Width of the window
037         */
038        public UI(int windowHeight, int windowWidth) {
039                // We call the parent's constructor
040                super();
041
042                Philophobia.getVerbose().calls("Creating UI class", "window/ui/UI.java", "UI.UI(int, int)");
043
044                this.windowHeight = windowHeight;
045                this.windowWidth = windowWidth;
046
047        }
048
049        /**
050         * UI class constructor
051         * @param windowHeight Height of the window
052         * @param windowWidth Width of the window
053         * @param world World to display
054         */
055        public UI(int windowHeight, int windowWidth, World world) {
056                super();
057
058                Philophobia.getVerbose().calls("Creating UI class", "window/ui/UI.java", "UI.UI(int, int,World)");
059
060                this.windowHeight = windowHeight;
061                this.windowWidth = windowWidth;
062                this.displayedWorld = world;
063        }
064        
065        /**
066         * Setter for the displayedWorld field
067         * @see #displayedWorld
068         * @param world World to display
069         */
070        public void setDisplayedWorld(World world) {
071                this.displayedWorld = world;
072        }
073
074        /**
075         * Function called when the program ask to paint the graphics
076         * @param g Graphics to use
077         */
078        public void paintComponent(Graphics g) {
079                Philophobia.getVerbose().calls("Painting UI components", "window/ui/UI.java", "UI.paintComponent(Graphics)");
080                super.paintComponent(g);
081                drawWorld(g);
082        }
083
084        /**
085         * Function called to draw the world
086         * <p>
087         * This function just display each Scenery one by one
088         * but trees are displayed after everything else as
089         * they must be on top of everything because of their
090         * size
091         * @see #displayedWorld
092         * @param g Graphics to use
093         */
094        private void drawWorld(Graphics g) {
095        
096                Philophobia.getVerbose().calls("Painting the world", "window/ui/UI.java", "UI.drawWorld(Graphics)");
097
098                Scenery[][] worldMap = displayedWorld.getMap();
099                int worldMapXSize = displayedWorld.getSizeX();
100                int worldMapYSize = displayedWorld.getSizeY();
101
102                for(int i = 0 ; i < worldMapXSize ; ++i) {
103                        for(int j = 0 ; j < worldMapYSize ; ++j) {
104                                if(!worldMap[i][j].getClass().getName().equals("Tree"))
105                                        worldMap[i][j].drawScenery(g, i*Scenery.getSceneryHeight(), j*Scenery.getSceneryWidth(), this);
106                        }
107                }
108
109                for(int i = 0 ; i < worldMapXSize ; ++i) {
110                        for(int j = 0 ; j < worldMapYSize ; ++j) {
111                                if(worldMap[i][j].getClass().getName().equals("Tree"))
112                                        worldMap[i][j].drawScenery(g, i*Scenery.getSceneryHeight(), j*Scenery.getSceneryWidth(), this);
113                        }
114                }
115
116        }
117
118};
