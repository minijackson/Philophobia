001package world.character;
002
003import main.Philophobia;
004import world.scenery.Scenery;
005import world.InteractiveObject;
006
007/**
008 * Class used to handle a character (player or non-player)
009 * <p>
010 * A character can be a purely decorative character
011 * or a talking character or a player character (the hero)
012 */
013abstract public class Character extends Scenery implements InteractiveObject {
014
015        /**
016         * Height of the character's image (fixed)
017         */
018        protected static final int CHARACTER_HEIGHT = 300;
019
020        /**
021         * Width of the character's image (fixed)
022         */
023        protected static final int CHARACTER_WIDTH = 170;
024
025        /**
026         * Horizontal position of the character
027         */
028        protected int posX;
029
030        /**
031         * Vertical position of the character
032         */
033        protected int posY;
034
035        /**
036         * Character class constructor
037         * @param imagePath Image filename representing the character
038         */
039        public Character(final String imagePath) {
040                Philophobia.getVerbose().calls("Character class created", "world/characters/Character.java", "Character.Character(String)");
041
042                super(image, CHARACTER_HEIGHT, CHARACTER_WIDTH);
043
044        }
045
046        /**
047         * Teleport the character horizontally regarding his current position
048         * @param distance Horizontal distance toward the character will be teleported
049         */
050        public void jumpX(int distance) {
051                if(distance != 0) {
052                        this.posX = this.posX + distance;
053                        this.draw();
054                }
055        }
056
057        /**
058         * Teleport the character vertically regarding his current position
059         * @param distance Vertical distance toward the character will be teleported
060         */
061        public void jumpY(int distance) {
062                if(distance != 0) {
063                        this.posY = this.posY + distance;
064                        this.draw();
065                }
066        }
067
068        /**
069         * Teleport the character regarding his current position
070         * @param distanceX Horizontal distance toward the character will be teleported
071         * @param distanceY Vertical distance toward the character will be teleported
072         */
073        public void jump(int distanceX, int distanceY) {
074                if(distanceX != 0 || distanceY != 0) {
075                        this.posX = this.posX + distance;
076                        this.posY = this.posY + distance;
077                        this.draw();
078                }
079        }
080
081        /**
082         * Slowly horizontally move the character to a certain point regarding his current position
083         * @param distance Horizontal distance toward the character will be moved
084         */
085        public void moveX(int distance) {
086                
087                if(distance != 0) {
088                        int step = (distance < 0)? -1 : 1; 
089
090                        this.jumpX(step);
091
092                        this.moveX(distance - step);
093                }
094
095        }
096
097        /**
098         * Slowly vertically move the character to a certain point regarding his current position
099         * @param distance Vertical distance toward the character will be moved
100         */
101        public void moveY(int distance) {
102
103                if(distance != 0) {
104                        int step = (distance < 0)? -1 : 1;
105
106                        this.jumpY(step);
107
108                        this.moveY(distance - step);
109                }
110
111        }
112
113        /**
114         * Slowly move the character to a certain point regarding his current position
115         * @param distanceX Horizontal distance toward the character will be moved
116         * @param distanceY Vertical distance toward the character will be moved
117         */
118        public void move(int distanceX, int distanceY) {
119                
120                if(distanceX != 0 || distanceY != 0) {
121                        int stepX = (distanceX < 0)? -1 : 1;
122                        int stepY = (distanceY < 0)? -1 : 1;
123
124                        this.jump(stepX, stepY);
125
126                        if(distanceX - stepX == 0) {
127                                moveY(distanceY - stepY);
128                                return;
129                        }
130
131                        if(distanceY - stepY == 0) {
132                                moveX(distanceX - stepX);
133                                return;
134                        }
135
136                        move(distanceX - stepX, distanceY - stepY);
137                }
138
139        }
140
141        // ===============================
142        // == InteractiveObject methods ==
143        // ===============================
144        
145        public void triggerAction() {
146
147        }
148
149        public void playerNear() {
150
151        }
152
153        public void playerEnter() {
154
155        }
156
157        public void playerLeave() {
158
159        }
160
161        public boolean isTraversable() {
162                return false;
163        }
164
165
166        
167};
