001package world.characters;
002
003import main.Philophobia;
004import world.Scenery;
005
006/**
007 * Class used to handle a character (player or non-player)
008 * <p>
009 * A character can be a purely decorative character
010 * or a talking character or a player character (the hero)
011 */
012abstract public class Character extends Scenery {
013
014        /**
015         * Height of the character's image (fixed)
016         */
017        protected static final int CHARACTER_HEIGHT = 300;
018
019        /**
020         * Width of the character's image (fixed)
021         */
022        protected static final int CHARACTER_WIDTH = 170;
023
024        /**
025         * Horizontal position of the character
026         */
027        protected int posX;
028
029        /**
030         * Vertical position of the character
031         */
032        protected int posY;
033
034        /**
035         * Boolean equals to true if the character is visible, false if the character is not
036         */
037        protected boolean visible;
038
039        /**
040         * Character class constructor
041         * @param image Image filename representing the character
042         */
043        public Character(final String image) {
044                Philophobia.getVerbose().calls("Character class created", "world/characters/Character.java", "Character.Character(String)");
045
046                this.visible = false;
047                super(image, CHARACTER_HEIGHT, CHARACTER_WIDTH);
048
049        }
050
051        /**
052         * Teleport the character horizontally regarding his current position
053         * @param distance Horizontal distance toward the character will be teleported
054         */
055        public void jumpX(int distance) {
056                if(distance != 0) {
057                        this.posX = this.posX + distance;
058                        this.draw();
059                }
060        }
061
062        /**
063         * Teleport the character vertically regarding his current position
064         * @param distance Vertical distance toward the character will be teleported
065         */
066        public void jumpY(int distance) {
067                if(distance != 0) {
068                        this.posY = this.posY + distance;
069                        this.draw();
070                }
071        }
072
073        /**
074         * Teleport the character regarding his current position
075         * @param distanceX Horizontal distance toward the character will be teleported
076         * @param distanceY Vertical distance toward the character will be teleported
077         */
078        public void jump(int distanceX, int distanceY) {
079                if(distanceX != 0 || distanceY != 0) {
080                        this.posX = this.posX + distance;
081                        this.posY = this.posY + distance;
082                        this.draw();
083                }
084        }
085
086        /**
087         * Slowly horizontally move the character to a certain point regarding his current position
088         * @param distance Horizontal distance toward the character will be moved
089         */
090        public void moveX(int distance) {
091                
092                if(distance != 0) {
093                        int step = (distance < 0)? -1 : 1; 
094
095                        this.jumpX(step);
096
097                        this.moveX(distance - step);
098                }
099
100        }
101
102        /**
103         * Slowly vertically move the character to a certain point regarding his current position
104         * @param distance Vertical distance toward the character will be moved
105         */
106        public void moveY(int distance) {
107
108                if(distance != 0) {
109                        int step = (distance < 0)? -1 : 1;
110
111                        this.jumpY(step);
112
113                        this.moveY(distance - step);
114                }
115
116        }
117
118        /**
119         * Slowly move the character to a certain point regarding his current position
120         * @param distanceX Horizontal distance toward the character will be moved
121         * @param distanceY Vertical distance toward the character will be moved
122         */
123        public void move(int distanceX, int distanceY) {
124                
125                if(distanceX != 0 || distanceY != 0) {
126                        int stepX = (distanceX < 0)? -1 : 1;
127                        int stepY = (distanceY < 0)? -1 : 1;
128
129                        this.jump(stepX, stepY);
130
131                        if(distanceX - stepX == 0) {
132                                moveY(distanceY - stepY);
133                                return;
134                        }
135
136                        if(distanceY - stepY == 0) {
137                                moveX(distanceX - stepX);
138                                return;
139                        }
140
141                        move(distanceX - stepX, distanceY - stepY);
142                }
143
144        }
145        
146};
