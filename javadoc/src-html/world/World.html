015/**
016 * World is a class used to handle the game world graphics
017 * <p>
018 * The World class use the Scenery objects (or the objects
019 * implementing the Scenery class) to display images of
020 * several objects corresponding to trees and traps and so on
021 */
022public class World {
023        
024        /**
025         * Horizontal size of the map
026         */
027        protected int sizeX;
028
029        /**
030         * Vertical size of the map
031         */
032        protected int sizeY;
033
034        /**
035         * Two-dimensional array containing all objects within the world
036         * <p>
037         * This object is filled using a file in the root directory
038         * and in the form of map[x][y] from top-left to right-bottom
039         */
040        protected Scenery[][] map;
041
042        /**
043         * World class constructor
044         * @param filename filename containing the data of the current map
045         */
046        public World(String filename) {
047                Philophobia.getVerbose().information("Creating World class", "world/World.java", "World.World(String)");
048
049                try {
050
051                        BufferedInputStream inputStream = new BufferedInputStream(new FileInputStream(new File(filename)));
052                        
053                        while(inputStream.read() != -1) {
054                                // Count
055                        }
056
057                        sizeX = 42;
058                        sizeY = 42;
059
060                        this.map = new Scenery[sizeX][sizeY];
061
062                        while(inputStream.read() != -1) {
063                                // Fill this.map
064                        }
065
066                } catch(FileNotFoundException e) {
067
068                        Philophobia.getVerbose().critical("File " + filename + " not found", "world/World.java", "World.World(String)");
069                        System.exit(-1);
070
071                } catch(IOException e) {
072                        Philophobia.getVerbose().serious("Error reading the file " + filename, "world/World.java", "World.World(String)");
073                }
074        }
075
076        /**
077         * Getter for the sizeX Field
078         * @return int Horizontal size of the map (in Scenery objects)
079         * @see #sizeX
080         */
081        public int getSizeX() {
082                return this.sizeX;
083        }
084
085        /**
086         * Getter for the sizeY Field
087         * @return int Vertical size of the map (in Scenery objects)
088         * @see #sizeY
089         */
090        public int getSizeY() {
091                return this.sizeY;
092        }
093
094        /**
095         * Getter for the map Field
096         * @return Scenery[][] The map of the world
097         * @see #map
098         */
099        public Scenery[][] getMap() {
100                return this.map;
101        }
102};
