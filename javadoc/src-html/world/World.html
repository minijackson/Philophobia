001package world;
002
003import main.Philophobia;
004import world.scenery.*;
005import world.character.*;
006
007import debug.Verbose;
008import javax.swing.JFrame;
009import java.awt.Graphics;
010
011/**
012 * World is a class used to handle the game world graphics
013 * <p>
014 * The World class use the Scenery objects (or the objects
015 * implementing the Scenery class) to display images of
016 * several objects corresponding to trees and traps and so on
017 */
018public class World {
019        
020        /**
021         * Horizontal size of the map
022         */
023        protected int sizeX;
024
025        /**
026         * Vertical size of the map
027         */
028        protected int sizeY;
029
030        /**
031         * Two-dimensional array containing all objects within the world
032         * <p>
033         * This object is filled using a file in the root directory
034         * and in the form of map[x][y] from top-left to right-bottom
035         */
036        protected Scenery[][] map;
037
038        /**
039         * Style of the world
040         */
041        protected String type;
042
043        /**
044         * World class constructor
045         * @param type Style of the world
046         */
047        public World(final String type) {
048                Philophobia.getVerbose().information("Creating World class", "world/World.java", "World.World(String)");
049
050                this.type = type;
051                this.sizeX = 128;
052                this.sizeY = 128;
053
054                generateWorld();
055        }
056
057        /**
058         * World class constructor with size parameters
059         * @param type Style of the world
060         * @param sizeX Horizontal size of the world
061         * @param sizeY Vertical size of the world
062         */
063        public World(final String type, final int sizeX, final int sizeY) {
064                Philophobia.getVerbose().information("Creating World class", "world/World.java", "World.World(String, int, int)");
065
066                this.type = type;
067                this.sizeX = sizeX;
068                this.sizeY = sizeY;
069
070                generateWorld();
071        }
072
073        /**
074         * Function randomly generating the world
075         */
076        protected void generateWorld() {
077                map = new Scenery[sizeX][sizeY];
078
079                // Loop generating the world
080                for(int i = 0 ; i < sizeX ; ++i) {
081                        for(int j = 0 ; i < sizeY ; ++j) {
082
083                                double random = Math.random();
084
085                                Scenery currentScenery;
086
087                                if(0 <= random && random < .3) {
088                                        currentScenery = new Tree(type);
089                                } else if(.3 <= random && random < .6) {
090                                        currentScenery = new Rock(type);
091                                } else {
092                                        currentScenery = new Ground(type);
093                                }
094
095                                map[i][j] = currentScenery;
096                        }
097                }
098        }
099
100        /**
101         * Getter for the sizeX Field
102         * @return int Horizontal size of the map (in Scenery objects)
103         * @see #sizeX
104         */
105        public int getSizeX() {
106                return this.sizeX;
107        }
108
109        /**
110         * Getter for the sizeY Field
111         * @return int Vertical size of the map (in Scenery objects)
112         * @see #sizeY
113         */
114        public int getSizeY() {
115                return this.sizeY;
116        }
117
118        /**
119         * Getter for the map Field
120         * @return Scenery[][] The map of the world
121         * @see #map
122         */
123        public Scenery[][] getMap() {
124                return this.map;
125        }
126};
