001package main;
002
003import java.util.Scanner;
004import debug.CliOptions;
005import debug.Verbose;
006
007import window.Window;
008
009/**
010 * Main class
011 */
012public class Philophobia {
013
014        /**
015         * Global variable used for debugging
016         * @see debug.Verbose
017         */
018        private static Verbose verbose;
019
020        /**
021         * Main window of the program
022         * @see window.Window
023         */
024        private static Window window;
025
026        /**
027         * Main function of the A3P program
028         * You can set options like verbose mode to the program
029         * @see debug.CliOptions
030         *
031         * @param args Global program options (see above)
032        */
033        public static void main(String[] args) {
034                
035                CliOptions options = new CliOptions(args);
036
037                Philophobia.verbose = new Verbose(options.getVerboseLevel());
038
039                if(!options.isHelpQueried()) {
040                        Philophobia.window = new Window();
041                } else {
042        
043                        System.out.println("");
044                        System.out.println("Usage: philophobia [options]");
045                        System.out.println("");
046                        System.out.println("Options:");
047                        System.out.println("  Debugging options:");
048                        System.out.println("    -v                    Activate verbose mode (to be use multiple times for more verbosity)");
049                        System.out.println("    -h      --help        Print this help");
050                        System.out.println("");
051
052                }
053                        
054        }
055
056        /**
057         * Getter for the verbose property
058         * @return The main class verbose static property
059         */
060        public static Verbose getVerbose() {
061                return verbose;
062        }
063
064};
