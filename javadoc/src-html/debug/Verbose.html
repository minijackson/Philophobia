001package debug;
002
003import java.util.logging.Logger;
004import java.util.logging.FileHandler;
005import java.util.logging.Level;
006import java.io.IOException;
007
008/**
009 * Class used to display and log messages all over this program
010 */
011public class Verbose {
012
013        /**
014         * Level of verbosing, from 0 to 5
015         * <p> 
016         * The number of information displayed and logged goes increasingly as the number increase
017         * - Level 1 : Only critical errors and exeptions are displayed on the console
018         * - Level 2 : Serious but not critical errors and level 1 displayed
019         * - Level 3 : Warning and level 2 displayed
020         * - Level 4 : Informations and level 3 displayed
021         * - Level 5 : Everything, included Class instanciations and function calls and level 4 displayed
022         */
023        protected int verboseMode = 0;
024        
025        /**
026         * Logger used to log messages into a file
027         */
028        Logger log;
029
030        /**
031         * Class constructor
032         * 
033         * @param level Level of "verbosing"
034         * @see #verboseMode
035         */
036        public Verbose(int level) {
037                this.verboseMode = level;
038
039                if(level > 0) {
040                        System.out.println("Verbose mode activated at level " + level);
041                        
042                                log = Logger.getLogger("Philophobia.log");
043                                
044                                try {
045                                        log.addHandler(new FileHandler("Philophobia.log"));
046                                } catch (IOException e) {
047                                        warning("Error initializing the log file", "org/debug/Verbose.java", "Verbose.Verbose(int)");
048                                }
049
050                                log.setLevel(Level.parse("ALL"));
051
052                                log.info("Verbose at level " + level + ".");
053                                log.info("Displayed messages are :");
054                                log.info("- Criticals");
055
056                                if(level >= 2) {
057                                        log.info("- Serious");
058                                }
059                                if(level >= 3) {
060                                        log.info("- Warnings");
061                                }
062                                if(level >= 4) {
063                                        log.info("- Informations");
064                                }
065                                if(level >= 5) {
066                                        log.info("- Class instanciations and function calls");
067                                }
068                }
069
070                this.calls("Verbose class created", "org/debug/Verbose.java", "Verbose.Verbose(int)");
071        }
072
073        /**
074         * Display and log a message with additionnal informations in the log file
075         * The message is displayed and logged if and only if the verbose level is greater or equal to 1
076         *
077         * @param message Message content
078         * @param file File in which this function is called
079         * @param location Class and function in which this function is called
080         */
081        public void critical(String message, String file, String location) {
082                if(verboseMode >= 1) {
083                        System.out.println("Critical : " + message);
084
085                        log.severe("Critical : " + message + " in file " + file + " in " + location);
086                }
087        }
088
089        /**
090         * Display and log a message with additionnal informations in the log file
091         * The message is displayed and logged if and only if the verbose level is greater or equal to 2
092         *
093         * @param message Message content
094         * @param file File in which this function is called
095     * @param location Class and function in which this function is called
096         */
097        public void serious(String message, String file, String location) {
098                if(verboseMode >= 2) {
099                        System.out.println("Serious : " + message);
100                        
101                        log.severe("Serious : " + message + " in file " + file + " in " + location);
102                }
103        }
104        
105        /**
106         * Display and log a message with additionnal informations in the log file
107         * The message is displayed and logged if and only if the verbose level is greater or equal to 3
108         *
109         * @param message Message content
110         * @param file File in which this function is called
111     * @param location Class and function in which this function is called
112         */
113        public void warning(String message, String file, String location) {
114                if(verboseMode >= 3) {
115                        System.out.println("Warning : " + message);
116                        
117                        log.warning("Warning : " + message + " in file " + file + " in " + location);
118                }
119        }
120        
121        /**
122         * Display and log a message with additionnal informations in the log file
123         * The message is displayed and logged if and only if the verbose level is greater or equal to 4
124         *
125         * @param message Message content
126         * @param file File in which this function is called
127     * @param location Class and function in which this function is called
128         */
129        public void information(String message, String file, String location) {
130                if(verboseMode >= 4) {
131                        System.out.println("Information : " + message);
132                        
133                        log.info("Information : " + message + " in file " + file + " in " + location);
134                }
135        }
136        
137        /**
138         * Display and log a message with additionnal informations in the log file
139         * The message is displayed and logged if and only if the verbose level is equal to 5
140         *
141         * @param message Message content
142         * @param file File in which this function is called
143     * @param location Class and function in which this function is called
144         */
145        public void calls(String message, String file, String location) {
146                if(verboseMode >= 5) {
147                        System.out.println("Calls : " + message);
148                        
149                        log.finer("Calls : " + message + " in file " + file + " in " + location);
150                }
151        }
152};
