001package debug;
002
003import java.util.List;
004import java.util.LinkedList;
005import java.util.ListIterator;
006
007/**
008 * Class used to analyze the commands passed to the program
009 * <p>
010 * For now, only the verbose "-v" can be passed to the program (once or several times)
011 * <p>
012 * You can set a level of verbose mode from 0 to 5, you just have to put -v for verbose mode level 1, -vv for level 2 and so on
013 */
014public class CliOptions {
015
016        /**
017         * Level of verbosing detected due to the args passed to the program
018         *
019         * @see Verbose
020         * @see Verbose#verboseMode
021         */
022        protected int verboseLevel;
023
024
025        /**
026         * Constructor of the CliOption
027         *
028         * @param args Options passed to the program when launching it
029         */
030        public CliOptions(String[] args) {
031
032                String oneDashOptions = new String();     // One dash options are letter by letter
033                List<String> twoDashesOptions = new LinkedList<String>(); // That is not the case with two dashes options which are word by word
034
035                // Check the arguments passed to the program
036                for(String str : args) {
037                        // Check if an option is set
038                        if(str.charAt(0) == '-' && str.length() > 1) {
039
040                                // This program support the fact that an option can be prfixed with either '-' or '--'
041                                if(str.charAt(1) == '-' && str.length() > 2) {
042                                        
043                                        twoDashesOptions.add(str.substring(2));
044
045                                } else {
046                                        
047                                        oneDashOptions += str.substring(1);
048
049                                }
050                                
051                                
052                        }
053                }
054
055                checkVerboseLevel(oneDashOptions);
056
057        }
058
059        /**
060         * Check the verbose level considering the options with one dash
061         * @param args The options with one dash passed to the program
062         */
063        private void checkVerboseLevel(String args) {
064                verboseLevel = 0;
065                
066                // Due to performance reasons, the length of the arguments are stored in a variable
067                int argsLength = args.length();
068
069                for(int i = 0 ; i < argsLength && verboseLevel < 5 ; ++i) {
070                        if(args.charAt(i) == 'v')
071                                ++verboseLevel;
072                }
073
074        }
075
076        /**
077         * Return the verbose level passed as options to the program
078         * @return The verbose level
079         */
080        public int getVerboseLevel() {
081                return verboseLevel;
082        }
083
084};
